### 1.3 核心实现逻辑

```python
def create_chat_model(name: str | None = None, thinking_enabled: bool = False, **kwargs) -> BaseChatModel:
    # 1. 从配置获取模型配置
    config = get_app_config()
    model_config = config.get_model_config(name)
    
    # 2. 通过反射动态加载模型类
    # "langchain_openai:ChatOpenAI" → 实际的类对象
    model_class = resolve_class(model_config.use, BaseChatModel)
    
    # 3. 处理 thinking 模式配置
    if thinking_enabled and model_config.supports_thinking:
        model_settings.update(model_config.when_thinking_enabled)
    
    # 4. 特殊处理 Codex 模型
    if issubclass(model_class, CodexChatModel):
        # Codex 不支持 max_tokens，映射到 reasoning_effort
        
    # 5. 创建模型实例
    model_instance = model_class(**kwargs, **model_settings)
    
    # 6. 挂载 LangSmith 追踪
    if is_tracing_enabled():
        model_instance.callbacks = [LangChainTracer(...)]
    
    return model_instance
```

### 1.4 特殊模型适配

| 模型 | 特殊处理 |
|------|---------|
| **OpenAI Codex** | 移除 max_tokens，映射 thinking 到 reasoning_effort |
| **Anthropic** | thinking 配置直接作为构造函数参数 |
| **DeepSeek** | patched_deepseek.py 修复兼容性问题 |
| **Minimax** | patched_minimax.py 修复 API 差异 |

### 2.3 Tool Search（延迟工具发现）

```
┌─────────────────────────────────────────────────────────────────────────────────────────┐
│                          Tool Search 延迟加载机制                                         │
├─────────────────────────────────────────────────────────────────────────────────────────┤
│                                                                                         │
│   问题：MCP 工具可能有几十个，一次性加载所有工具描述会消耗大量 Token                         │
│                                                                                         │
│   解决方案：Tool Search + Deferred Registry                                             │
│                                                                                         │
│   ┌───────────────────────────────────────────────────────────────────────────────────┐ │
│   │  1. 工具加载时：                                                                    │ │
│   │     if tool_search.enabled:                                                       │ │
│   │         MCP tools → DeferredToolRegistry (注册但不暴露给 LLM)                       │ │
│   │         builtin tools += [tool_search]                                            │ │
│   │                                                                                   │ │
│   │  2. DeferredToolFilterMiddleware:                                                 │ │
│   │     wrap_model_call() 中移除延迟工具的 schema                                       │ │
│   │     LLM.bind_tools() 只收到活跃工具的 schema                                       │ │
│   │                                                                                   │ │
│   │  3. Agent 运行时：                                                                 │ │
│   │     LLM 调用 tool_search(query="github")                                          │ │
│   │     → 搜索 DeferredToolRegistry                                                   │ │
│   │     → 返回匹配工具的 schema                                                        │ │
│   │     → 下一轮 LLM 可以调用该工具                                                    │ │
│   └───────────────────────────────────────────────────────────────────────────────────┘ │
│                                                                                         │
│   优势：节省 Token + 按需发现工具                                                        │
│                                                                                         │
└─────────────────────────────────────────────────────────────────────────────────────────┘
```

def apply_prompt_template(available_skills: set[str]) -> str:

    # 构建技能指令部分
    skills_section = ""
    for skill in skills:

    return f"""

# sandbox/middleware.py
class SandboxMiddleware(AgentMiddleware):
    def before_agent(self, state, runtime):
        thread_id = runtime.context.get("thread_id")
        provider = get_sandbox_provider()
        
        # 获取或创建沙箱
        sandbox_id = provider.acquire(thread_id)
        sandbox = provider.get(sandbox_id)
        
        # 注入到 state
        return {
            "sandbox": {
                "id": sandbox_id,
                "workspace_path": "/mnt/user-data/workspace",
                "uploads_path": "/mnt/user-data/uploads",

        }

            # 移除 <uploaded_files> 块（会话级临时数据）

        elif msg.type == "ai":
